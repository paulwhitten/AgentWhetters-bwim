\documentclass[border=10pt]{standalone}
\usepackage{tikz}
\usetikzlibrary{3d,calc,arrows.meta}

\pgfmathsetmacro{\blocksize}{0.55}

\begin{document}
\begin{tikzpicture}[>=Stealth]

% ============================================================
% 2D PLAN — flat horizontal grid ABOVE the 3D space
% ============================================================
\begin{scope}[shift={(0, 0)}]
    \pgfmathsetmacro{\cs}{0.38} % cell size

    % Grid lines — 9x9
    \foreach \i in {0,...,9} {
        \draw[gray!35, very thin] (\i*\cs, 0) -- (\i*\cs, 9*\cs);
        \draw[gray!35, very thin] (0, \i*\cs) -- (9*\cs, \i*\cs);
    }
    % Border
    \draw[black!70, semithick] (0,0) rectangle (9*\cs, 9*\cs);

    % Red cells (footprint) — x=4,5,6 at z=5
    \fill[red!50!white, draw=red!60!black, semithick]
        (4*\cs, 5*\cs) rectangle (5*\cs, 6*\cs);
    \node[font=\tiny\bfseries, red!60!black] at (4.5*\cs, 5.5*\cs) {3};

    \fill[red!50!white, draw=red!60!black, semithick]
        (5*\cs, 5*\cs) rectangle (6*\cs, 6*\cs);
    \node[font=\tiny\bfseries, red!60!black] at (5.5*\cs, 5.5*\cs) {1};

    \fill[red!50!white, draw=red!60!black, semithick]
        (6*\cs, 5*\cs) rectangle (7*\cs, 6*\cs);
    \node[font=\tiny\bfseries, red!60!black] at (6.5*\cs, 5.5*\cs) {1};

    % Blue cell (drop target) — x=6, z=3
    \fill[blue!40!white, draw=blue!60!black, semithick]
        (6*\cs, 3*\cs) rectangle (7*\cs, 4*\cs);
    \node[font=\tiny\bfseries, blue!60!black] at (6.5*\cs, 3.5*\cs) {1};

    % Axis labels
    \draw[->, thick, black!70] (-0.3, -0.3) -- (9*\cs + 0.6, -0.3)
        node[right, font=\small] {$x$};
    \draw[->, thick, black!70] (-0.3, -0.3) -- (-0.3, 9*\cs + 0.6)
        node[above, font=\small] {$z$};

    % Label above
    \node[font=\sffamily\bfseries] at (4.5*\cs, 9*\cs + 0.9)
        {2D Plan $(x, z)$};

    % Mark the blue cell for the connecting arrow
    \coordinate (bluecell) at (6.5*\cs, 3.5*\cs);
    \coordinate (bluecellbottom) at (6.5*\cs, 3*\cs);
\end{scope}

% ============================================================
% 3D BUILD SPACE — below the 2D plan
% ============================================================
\begin{scope}[shift={(0.5, -3.3)},
    x={(0.55cm, -0.28cm)},
    y={(0cm, 0.7cm)},
    z={(-0.55cm, -0.28cm)},
]

    % Helper: Draw a cube at position (bx, by, bz) with color
    \newcommand{\block}[4]{% x, y, z, color
        \pgfmathsetmacro{\bx}{#1*\blocksize}
        \pgfmathsetmacro{\by}{#2*\blocksize}
        \pgfmathsetmacro{\bz}{#3*\blocksize}
        \pgfmathsetmacro{\bs}{\blocksize}
        % Top face (lightest)
        \fill[#4!50!white, draw=#4!60!black, line width=0.3pt]
            (\bx, \by+\bs, \bz) --
            (\bx+\bs, \by+\bs, \bz) --
            (\bx+\bs, \by+\bs, \bz+\bs) --
            (\bx, \by+\bs, \bz+\bs) -- cycle;
        % Front face (medium)
        \fill[#4!70!white, draw=#4!60!black, line width=0.3pt]
            (\bx, \by, \bz+\bs) --
            (\bx+\bs, \by, \bz+\bs) --
            (\bx+\bs, \by+\bs, \bz+\bs) --
            (\bx, \by+\bs, \bz+\bs) -- cycle;
        % Right face (darkest)
        \fill[#4!40!black, draw=#4!60!black, line width=0.3pt]
            (\bx+\bs, \by, \bz) --
            (\bx+\bs, \by, \bz+\bs) --
            (\bx+\bs, \by+\bs, \bz+\bs) --
            (\bx+\bs, \by+\bs, \bz) -- cycle;
    }

    \pgfmathsetmacro{\gs}{9*\blocksize}

    % Ground plane grid — 9x9
    \foreach \i in {0,...,9} {
        \pgfmathsetmacro{\pos}{\i*\blocksize}
        \draw[gray!30, very thin] (\pos, 0, 0) -- (\pos, 0, \gs);
        \draw[gray!30, very thin] (0, 0, \pos) -- (\gs, 0, \pos);
    }
    \draw[gray!60, thin] (0,0,0) -- (\gs,0,0) -- (\gs,0,\gs) -- (0,0,\gs) -- cycle;

    % Red block stack at (x=4, z=5) — 3 blocks tall
    \block{4}{0}{5}{red}
    \block{4}{1}{5}{red}
    \block{4}{2}{5}{red}

    % Red blocks along z=5
    \block{5}{0}{5}{red}
    \block{6}{0}{5}{red}

    % Blue block column guides (dashed vertical lines)
    \pgfmathsetmacro{\bcx}{6*\blocksize}
    \pgfmathsetmacro{\bcz}{3*\blocksize}
    \draw[blue!40, dashed, thin]
        (\bcx, 0, \bcz) -- (\bcx, 5*\blocksize, \bcz);
    \draw[blue!40, dashed, thin]
        (\bcx+\blocksize, 0, \bcz) -- (\bcx+\blocksize, 5*\blocksize, \bcz);
    \draw[blue!40, dashed, thin]
        (\bcx, 0, \bcz+\blocksize) -- (\bcx, 5*\blocksize, \bcz+\blocksize);
    \draw[blue!40, dashed, thin]
        (\bcx+\blocksize, 0, \bcz+\blocksize) -- (\bcx+\blocksize, 5*\blocksize, \bcz+\blocksize);

    % Dropping blue block (elevated, mid-fall)
    \block{6}{4}{3}{blue}

    % Downward arrow showing the drop
    \draw[->, blue!70!black, line width=1.2pt]
        (6*\blocksize + 0.5*\blocksize, 3.7*\blocksize, 3*\blocksize + 0.5*\blocksize) --
        (6*\blocksize + 0.5*\blocksize, 0.6*\blocksize, 3*\blocksize + 0.5*\blocksize);

    % Axes
    \draw[->, thick, black!70] (0,0,0) -- (10.5*\blocksize, 0, 0)
        node[right, font=\small] {$x$};
    \draw[->, thick, black!70] (0,0,0) -- (0, 6*\blocksize, 0)
        node[above, font=\small] {$y$};
    \draw[->, thick, black!70] (0,0,0) -- (0, 0, 10.5*\blocksize)
        node[left, font=\small] {$z$};

    % y-axis tick marks
    \foreach \i in {1,...,5} {
        \pgfmathsetmacro{\ypos}{\i*\blocksize}
        \draw[gray!50, thin] (-0.1, \ypos, 0) -- (0.1, \ypos, 0);
    }

    % Label
    \node[font=\sffamily\bfseries, anchor=south] at (4.5*\blocksize, 6.5*\blocksize, 0)
        {3D Build Space};

    % Save coordinate at top of blue column for connecting arrow
    \coordinate (bluetop) at (6*\blocksize + 0.5*\blocksize, 5*\blocksize, 3*\blocksize + 0.5*\blocksize);

\end{scope}

% ============================================================
% CONNECTING ARROW: blue cell in 2D plan → top of column in 3D space
% ============================================================
\draw[->, thick, blue!60!black, dashed, line width=1.2pt]
    (bluecellbottom) -- (bluetop)
    node[midway, right, font=\scriptsize\sffamily, xshift=2pt] {place block};

% ============================================================
% BUILD SEQUENCE — right side, vertically centered
% ============================================================
\node[anchor=north west] at (5.5, -2.0) {%
\begin{tikzpicture}
    \node[font=\sffamily\bfseries\large, anchor=north west] at (0, 0) {Build Sequence};
    \node[font=\small, anchor=north west, text width=4.2cm] at (0, -0.7) {
        \begin{enumerate}
            \setlength{\itemsep}{4pt}
            \item[\textcolor{red!70!black}{\textbf{1.}}] \textcolor{red!70!black}{Stack 3 red at $(5,6)$}
            \item[\textcolor{red!70!black}{\textbf{2.}}] \textcolor{red!70!black}{Place red at $(6,6)$}
            \item[\textcolor{red!70!black}{\textbf{3.}}] \textcolor{red!70!black}{Place red at $(7,6)$}
            \item[\textcolor{blue!70!black}{\textbf{4.}}] \textcolor{blue!70!black}{Place blue at $(7,4)$}
        \end{enumerate}
    };
\end{tikzpicture}%
};

\end{tikzpicture}
\end{document}
