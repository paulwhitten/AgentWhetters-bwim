% ============================================================
% Shared block-drawing macro
% ============================================================
\newcommand{\block}[4]{% x, y, z, color
    \pgfmathsetmacro{\bx}{#1*\blocksize}
    \pgfmathsetmacro{\by}{#2*\blocksize}
    \pgfmathsetmacro{\bz}{#3*\blocksize}
    \pgfmathsetmacro{\bs}{\blocksize}
    % Top face (lightest)
    \fill[#4!50!white, draw=#4!60!black, line width=0.3pt]
        (\bx, \by+\bs, \bz) --
        (\bx+\bs, \by+\bs, \bz) --
        (\bx+\bs, \by+\bs, \bz+\bs) --
        (\bx, \by+\bs, \bz+\bs) -- cycle;
    % Front face (medium)
    \fill[#4!70!white, draw=#4!60!black, line width=0.3pt]
        (\bx, \by, \bz+\bs) --
        (\bx+\bs, \by, \bz+\bs) --
        (\bx+\bs, \by+\bs, \bz+\bs) --
        (\bx, \by+\bs, \bz+\bs) -- cycle;
    % Right face (darkest)
    \fill[#4!40!black, draw=#4!60!black, line width=0.3pt]
        (\bx+\bs, \by, \bz) --
        (\bx+\bs, \by, \bz+\bs) --
        (\bx+\bs, \by+\bs, \bz+\bs) --
        (\bx+\bs, \by+\bs, \bz) -- cycle;
}
